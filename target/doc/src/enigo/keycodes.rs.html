622        use xkeysym::Keysym;
623
624        trace!("Key::from(key: {key:?})");
625
626        #[allow(clippy::match_same_arms)]
627        match key {
628            Key::Unicode(c) => xkeysym::Keysym::from_char(c),
629            Key::Alt | Key::Option => Keysym::Alt_L,
630            Key::Backspace => Keysym::BackSpace,
631            Key::Begin => Keysym::Begin,
632            Key::Break => Keysym::Break,
633            Key::Cancel => Keysym::Cancel,
634            Key::CapsLock => Keysym::Caps_Lock,
635            Key::Clear => Keysym::Clear,
636            Key::Control | Key::LControl => Keysym::Control_L,
637            Key::Delete => Keysym::Delete,
638            Key::DownArrow => Keysym::Down,
639            Key::End => Keysym::End,
640            Key::Escape => Keysym::Escape,
641            Key::Execute => Keysym::Execute,
642            Key::F1 => Keysym::F1,
643            Key::F2 => Keysym::F2,
644            Key::F3 => Keysym::F3,

648            Key::F7 => Keysym::F7,
649            Key::F8 => Keysym::F8,
650            Key::F9 => Keysym::F9,
651            Key::F10 => Keysym::F10,
652            Key::F11 => Keysym::F11,
653            Key::F12 => Keysym::F12,
